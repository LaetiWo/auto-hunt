Ah oui, excellente question ! Avec Resend, on peut envoyer des emails en plus des notifications dans l'app. Voici comment intégrer Resend :

1. Installation de Resend
bash
npm install resend
2. Configuration Resend
typescript
// lib/resend.ts
import { Resend } from 'resend';

export const resend = new Resend(process.env.RESEND_API_KEY);
Ajoutez dans votre .env.local :

RESEND_API_KEY=re_xxxxxxxxxxxx

// lib/email-service.ts
import { resend } from './resend';
import { RentalRequestEmail } from '@/emails/rental-request-email';
import { PurchaseInquiryEmail } from '@/emails/purchase-inquiry-email';
import { RequestConfirmedEmail } from '@/emails/request-confirmed-email';

const FROM_EMAIL = 'notifications@votredomaine.com'; // Remplacez par votre domaine vérifié

export class EmailService {
  
  // Email pour demande de location
  static async sendRentalRequestEmail(data: {
    to: string;
    ownerName: string;
    requesterName: string;
    requesterEmail: string;
    requesterPhone?: string;
    vehicleTitle: string;
    vehicleImage?: string;
    message: string;
    startDate?: string;
    endDate?: string;
    requestUrl: string;
  }) {
    try {
      await resend.emails.send({
        from: FROM_EMAIL,
        to: data.to,
        subject: `Nouvelle demande de location - ${data.vehicleTitle}`,
        react: RentalRequestEmail({
          ownerName: data.ownerName,
          requesterName: data.requesterName,
          requesterEmail: data.requesterEmail,
          requesterPhone: data.requesterPhone,
          vehicleTitle: data.vehicleTitle,
          vehicleImage: data.vehicleImage,
          message: data.message,
          startDate: data.startDate,
          endDate: data.endDate,
          requestUrl: data.requestUrl,
        }),
      });
    } catch (error) {
      console.error('Error sending rental request email:', error);
      // Ne pas bloquer si l'email échoue
    }
  }

  // Email pour demande d'achat/contact
  static async sendPurchaseInquiryEmail(data: {
    to: string;
    ownerName: string;
    requesterName: string;
    requesterEmail: string;
    requesterPhone?: string;
    vehicleTitle: string;
    vehicleImage?: string;
    message: string;
    requestUrl: string;
  }) {
    try {
      await resend.emails.send({
        from: FROM_EMAIL,
        to: data.to,
        subject: `Nouveau contact pour ${data.vehicleTitle}`,
        react: PurchaseInquiryEmail({
          ownerName: data.ownerName,
          requesterName: data.requesterName,
          requesterEmail: data.requesterEmail,
          requesterPhone: data.requesterPhone,
          vehicleTitle: data.vehicleTitle,
          vehicleImage: data.vehicleImage,
          message: data.message,
          requestUrl: data.requestUrl,
        }),
      });
    } catch (error) {
      console.error('Error sending purchase inquiry email:', error);
    }
  }

  // Email de confirmation de demande
  static async sendRequestConfirmedEmail(data: {
    to: string;
    requesterName: string;
    ownerName: string;
    ownerEmail: string;
    ownerPhone?: string;
    vehicleTitle: string;
    vehicleImage?: string;
    type: 'RENTAL' | 'PURCHASE';
    startDate?: string;
    endDate?: string;
    dashboardUrl: string;
  }) {
    try {
      await resend.emails.send({
        from: FROM_EMAIL,
        to: data.to,
        subject: `✅ Demande confirmée - ${data.vehicleTitle}`,
        react: RequestConfirmedEmail({
          requesterName: data.requesterName,
          ownerName: data.ownerName,
          ownerEmail: data.ownerEmail,
          ownerPhone: data.ownerPhone,
          vehicleTitle: data.vehicleTitle,
          vehicleImage: data.vehicleImage,
          type: data.type,
          startDate: data.startDate,
          endDate: data.endDate,
          dashboardUrl: data.dashboardUrl,
        }),
      });
    } catch (error) {
      console.error('Error sending confirmation email:', error);
    }
  }
}

// lib/request-service.ts (version mise à jour)
import { databases } from "@/lib/appwrite";
import { ID, Query } from "appwrite";
import { NotificationService } from "./notification-service";
import { EmailService } from "./email-service";

const DATABASE_ID = process.env.NEXT_PUBLIC_APPWRITE_DATABASE_ID!;
const REQUESTS_COLLECTION_ID = "vehicle_requests";

export class VehicleRequestService {
  
  static async createRequest(data: {
    vehicleId: string;
    vehicleTitle: string;
    vehicleImage?: string;
    ownerId: string;
    ownerName: string;
    ownerEmail: string; // Ajouté
    requesterId: string;
    requesterName: string;
    requesterEmail: string;
    requesterPhone?: string;
    type: "RENTAL" | "PURCHASE";
    message: string;
    startDate?: string;
    endDate?: string;
  }) {
    try {
      const request = await databases.createDocument(
        DATABASE_ID,
        REQUESTS_COLLECTION_ID,
        ID.unique(),
        {
          ...data,
          status: "PENDING",
          createdAt: new Date().toISOString(),
        }
      );

      const notificationMessage = data.type === "RENTAL"
        ? `${data.requesterName} souhaite louer votre véhicule "${data.vehicleTitle}"`
        : `${data.requesterName} est intéressé par votre véhicule "${data.vehicleTitle}"`;

      // Notification dans l'app
      await NotificationService.createNotification({
        userId: data.ownerId,
        type: data.type === "RENTAL" ? "RENTAL_REQUEST" : "PURCHASE_INQUIRY",
        title: data.type === "RENTAL" ? "Nouvelle demande de location" : "Nouveau contact",
        message: notificationMessage,
        vehicleId: data.vehicleId,
        senderId: data.requesterId,
        senderName: data.requesterName,
        requestId: request.$id,
      });

      // Email au propriétaire
      const requestUrl = `${process.env.NEXT_PUBLIC_APP_URL}/dashboard/requests/${request.$id}`;
      
      if (data.type === "RENTAL") {
        await EmailService.sendRentalRequestEmail({
          to: data.ownerEmail,
          ownerName: data.ownerName,
          requesterName: data.requesterName,
          requesterEmail: data.requesterEmail,
          requesterPhone: data.requesterPhone,
          vehicleTitle: data.vehicleTitle,
          vehicleImage: data.vehicleImage,
          message: data.message,
          startDate: data.startDate,
          endDate: data.endDate,
          requestUrl,
        });
      } else {
        await EmailService.sendPurchaseInquiryEmail({
          to: data.ownerEmail,
          ownerName: data.ownerName,
          requesterName: data.requesterName,
          requesterEmail: data.requesterEmail,
          requesterPhone: data.requesterPhone,
          vehicleTitle: data.vehicleTitle,
          vehicleImage: data.vehicleImage,
          message: data.message,
          requestUrl,
        });
      }

      return request;
    } catch (error) {
      console.error("Error creating request:", error);
      throw error;
    }
  }

  static async confirmRequest(requestId: string, ownerId: string) {
    try {
      const request = await databases.getDocument(
        DATABASE_ID,
        REQUESTS_COLLECTION_ID,
        requestId
      );

      if (request.ownerId !== ownerId) {
        throw new Error("Unauthorized");
      }

      const updatedRequest = await databases.updateDocument(
        DATABASE_ID,
        REQUESTS_COLLECTION_ID,
        requestId,
        { status: "CONFIRMED" }
      );

      const notificationMessage = request.type === "RENTAL"
        ? `${request.ownerName} a accepté votre demande de location pour "${request.vehicleTitle}"`
        : `${request.ownerName} a confirmé votre demande pour "${request.vehicleTitle}"`;

      // Notification dans l'app
      await NotificationService.createNotification({
        userId: request.requesterId,
        type: "REQUEST_CONFIRMED",
        title: "Demande confirmée ✅",
        message: notificationMessage,
        vehicleId: request.vehicleId,
        senderId: ownerId,
        senderName: request.ownerName,
        requestId: requestId,
      });

      // Email de confirmation au demandeur
      const dashboardUrl = `${process.env.NEXT_PUBLIC_APP_URL}/dashboard/my-requests`;
      
      await EmailService.sendRequestConfirmedEmail({
        to: request.requesterEmail,
        requesterName: request.requesterName,
        ownerName: request.ownerName,
        ownerEmail: request.ownerEmail,
        ownerPhone: request.ownerPhone,
        vehicleTitle: request.vehicleTitle,
        vehicleImage: request.vehicleImage,
        type: request.type,
        startDate: request.startDate,
        endDate: request.endDate,
        dashboardUrl,
      });

      return updatedRequest;
    } catch (error) {
      console.error("Error confirming request:", error);
      throw error;
    }
  }

  // ... reste du code identique
}
